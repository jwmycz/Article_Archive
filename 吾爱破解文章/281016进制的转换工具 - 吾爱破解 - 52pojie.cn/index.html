哈喽！友友们，小弟这次来分享一个简单实用的命令行进制转换工具，支持二进制、八进制、十进制、十六进制之间的相互转换{:1_918:}

代码如下：

"""
进制转换工具 v1.0
支持2/8/10/16进制互转，带历史记录持久化
作者：@HJY
"""

import os
import json
import tkinter as tk
from tkinter import ttk, messagebox
from datetime import datetime
from typing import Dict, Any, List, Optional

class BaseConverter:
    """进制转换核心处理类"""
    BASE_NAMES = {2: "二进制", 8: "八进制", 10: "十进制", 16: "十六进制"}
    BASE_PREFIXES = {2: "0b", 8: "0o", 10: "", 16: "0x"}

    def __init__(self):
        self.max_history = 10
        self.history_path = os.path.join(os.path.expanduser("~"), ".base_converter_history.json")
        self.history = self._load_history()

    def validate_input(self, value: str, base: int) -> bool:
        """验证输入是否符合指定进制格式"""
        if not value or value.isspace():
            return False

        raw_value = self._strip_prefix(value, base)
        try:
            int(raw_value, base)
            return True
        except ValueError:
            return False

    def _strip_prefix(self, value: str, base: int) -> str:
        """移除进制前缀"""
        val = value.lower().strip()
        prefix_map = {2: "0b", 8: "0o", 16: "0x"}

        if base in prefix_map and val.startswith(prefix_map[base]):
            return val[2:]
        return val

    def convert(self, value: str, from_base: int) -> Optional[Dict[int, str]]:
        """执行进制转换，返回其他进制结果"""
        if not self.validate_input(value, from_base):
            return None

        raw_value = self._strip_prefix(value, from_base)
        try:
            dec_val = int(raw_value, from_base)
            results = {
                2: bin(dec_val)[2:],
                8: oct(dec_val)[2:],
                10: str(dec_val),
                16: hex(dec_val)[2:].upper()
            }
            self._append_history(from_base, raw_value, dec_val)
            return results
        except Exception as e:
            print(f"转换失败: {str(e)}")
            return None

    def _append_history(self, from_base: int, input_val: str, dec_val: int) -> None:
        """追加转换记录到历史（去重+限制数量）"""
        # 去重逻辑
        for item in self.history:
            if item["from_base"] == from_base and item["input_value"] == input_val:
                return

        timestamp = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        history_item = {
            "from_base": from_base,
            "from_base_name": self.BASE_NAMES[from_base],
            "input_value": input_val,
            "results": {
                2: bin(dec_val)[2:],
                8: oct(dec_val)[2:],
                10: str(dec_val),
                16: hex(dec_val)[2:].upper()
            },
            "timestamp": timestamp
        }

        self.history.insert(0, history_item)
        self.history = self.history[:self.max_history]
        self._save_history()

    def _load_history(self) -> List[Dict[str, Any]]:
        """加载本地历史记录"""
        if not os.path.exists(self.history_path):
            return []

        try:
            with open(self.history_path, 'r', encoding='utf-8') as f:
                return json.load(f)
        except (json.JSONDecodeError, IOError, PermissionError) as e:
            print(f"历史记录加载失败: {e}")
            return []

    def _save_history(self) -> None:
        """保存历史记录到本地文件"""
        try:
            with open(self.history_path, 'w', encoding='utf-8') as f:
                json.dump(self.history, f, ensure_ascii=False, indent=2)
        except (IOError, PermissionError) as e:
            print(f"历史记录保存失败: {e}")

    def clear_history(self) -> None:
        """清空所有历史记录"""
        self.history = []
        self._save_history()

    def get_history(self) -> List[Dict[str, Any]]:
        """获取所有历史记录"""
        return self.history

class BaseConverterGUI:
    """图形界面类 - 与HTML界面风格一致"""
    def __init__(self, root):
        self.root = root
        self.root.title("进制转换工具 v1.0")
        self.root.geometry("600x450")
        self.root.resizable(False, False)

        # 初始化转换器
        self.converter = BaseConverter()

        # 设置界面样式
        self._setup_style()

        # 构建界面布局
        self._create_widgets()

    def _setup_style(self):
        """设置界面样式（简洁大方）"""
        self.root.configure(bg="#f0f0f0")
        self.style = ttk.Style()
        self.style.configure("TLabel", font=("Microsoft YaHei", 10), background="#f0f0f0")
        self.style.configure("TButton", font=("Microsoft YaHei", 10), padding=5)
        self.style.configure("TCombobox", font=("Microsoft YaHei", 10))

    def _create_widgets(self):
        """创建所有界面组件"""
        # 标题区域
        title_label = tk.Label(
            self.root,
            text="进制转换工具",
            font=("Microsoft YaHei", 16, "bold"),
            bg="#f0f0f0",
            fg="#333333"
        )
        title_label.pack(pady=10)

        # 输入区域框架
        input_frame = tk.Frame(self.root, bg="#f0f0f0")
        input_frame.pack(pady=5, padx=20, fill=tk.X)

        # 源进制选择
        self.base_var = tk.StringVar(value="十进制")
        base_label = tk.Label(input_frame, text="源进制：", bg="#f0f0f0", font=("Microsoft YaHei", 10))
        base_label.grid(row=0, column=0, padx=5, sticky=tk.W)

        base_combobox = ttk.Combobox(
            input_frame,
            textvariable=self.base_var,
            values=["二进制", "八进制", "十进制", "十六进制"],
            state="readonly",
            width=10
        )
        base_combobox.grid(row=0, column=1, padx=5)

        # 输入框
        self.input_entry = tk.Entry(
            input_frame,
            font=("Microsoft YaHei", 10),
            width=30,
            relief=tk.SOLID,
            bd=1
        )
        self.input_entry.grid(row=0, column=2, padx=5)
        self.input_entry.bind("<Return>", lambda e: self._on_convert())  # 回车触发转换

        # 转换按钮
        convert_btn = tk.Button(
            input_frame,
            text="转换",
            font=("Microsoft YaHei", 10),
            bg="#4CAF50",
            fg="white",
            relief=tk.FLAT,
            padx=15,
            command=self._on_convert
        )
        convert_btn.grid(row=0, column=3, padx=5)

        # 结果展示区域框架
        result_frame = tk.Frame(self.root, bg="#f0f0f0")
        result_frame.pack(pady=10, padx=20, fill=tk.X)

        # 结果标签和输出框
        self.result_entries = {}
        bases = [("二进制", 2), ("八进制", 8), ("十进制", 10), ("十六进制", 16)]

        for idx, (name, base) in enumerate(bases):
            # 行布局：标签 + 输出框
            label = tk.Label(result_frame, text=f"{name}：", bg="#f0f0f0", font=("Microsoft YaHei", 10))
            label.grid(row=idx, column=0, padx=5, pady=3, sticky=tk.W)

            entry = tk.Entry(
                result_frame,
                font=("Microsoft YaHei", 10),
                width=50,
                relief=tk.SOLID,
                bd=1,
                state="readonly"
            )
            entry.grid(row=idx, column=1, padx=5, pady=3, sticky=tk.W)
            self.result_entries[base] = entry

        # 功能按钮区域
        btn_frame = tk.Frame(self.root, bg="#f0f0f0")
        btn_frame.pack(pady=15)

        # 清空按钮
        clear_btn = tk.Button(
            btn_frame,
            text="清空输入",
            font=("Microsoft YaHei", 10),
            bg="#f44336",
            fg="white",
            relief=tk.FLAT,
            padx=15,
            pady=5,
            command=self._on_clear
        )
        clear_btn.grid(row=0, column=0, padx=8)

        # 历史记录按钮
        history_btn = tk.Button(
            btn_frame,
            text="历史记录",
            font=("Microsoft YaHei", 10),
            bg="#2196F3",
            fg="white",
            relief=tk.FLAT,
            padx=15,
            pady=5,
            command=self._show_history
        )
        history_btn.grid(row=0, column=1, padx=8)

        # 清除历史按钮
        clear_history_btn = tk.Button(
            btn_frame,
            text="清除历史",
            font=("Microsoft YaHei", 10),
            bg="#FF9800",
            fg="white",
            relief=tk.FLAT,
            padx=15,
            pady=5,
            command=self._clear_history
        )
        clear_history_btn.grid(row=0, column=2, padx=8)

    def _get_selected_base(self) -> int:
        """获取选中的源进制对应的数值"""
        base_name = self.base_var.get()
        for base, name in self.converter.BASE_NAMES.items():
            if name == base_name:
                return base
        return 10  # 默认十进制

    def _on_convert(self):
        """转换按钮点击事件"""
        input_value = self.input_entry.get().strip()
        if not input_value:
            messagebox.showwarning("提示", "请输入要转换的数值！")
            return

        from_base = self._get_selected_base()

        # 验证并转换
        if not self.converter.validate_input(input_value, from_base):
            messagebox.showerror("错误", f"输入的不是有效的{self.converter.BASE_NAMES[from_base]}数值！")
            return

        results = self.converter.convert(input_value, from_base)
        if results:
            # 更新结果框
            for base, entry in self.result_entries.items():
                entry.config(state="normal")
                entry.delete(0, tk.END)
                # 源进制显示带前缀，其他进制正常显示
                prefix = self.converter.BASE_PREFIXES[base] if base == from_base else ""
                entry.insert(0, f"{prefix}{results[base]}")
                entry.config(state="readonly")

    def _on_clear(self):
        """清空输入按钮点击事件"""
        self.input_entry.delete(0, tk.END)
        # 清空结果框
        for entry in self.result_entries.values():
            entry.config(state="normal")
            entry.delete(0, tk.END)
            entry.config(state="readonly")

    def _show_history(self):
        """显示历史记录弹窗"""
        history = self.converter.get_history()
        if not history:
            messagebox.showinfo("历史记录", "暂无转换记录！")
            return

        # 创建历史记录窗口
        history_win = tk.Toplevel(self.root)
        history_win.title("历史记录")
        history_win.geometry("550x400")
        history_win.resizable(False, False)
        history_win.configure(bg="#f0f0f0")

        # 创建滚动文本框
        text = tk.Text(
            history_win,
            font=("Microsoft YaHei", 9),
            wrap=tk.WORD,
            relief=tk.SOLID,
            bd=1
        )
        scrollbar = ttk.Scrollbar(history_win, orient=tk.VERTICAL, command=text.yview)
        text.configure(yscrollcommand=scrollbar.set)

        text.pack(side=tk.LEFT, fill=tk.BOTH, expand=True, padx=10, pady=10)
        scrollbar.pack(side=tk.RIGHT, fill=tk.Y, padx=(0, 10), pady=10)

        # 填充历史记录
        text.insert(tk.END, "=== 进制转换历史记录 ===\n\n")
        for idx, item in enumerate(history, 1):
            text.insert(tk.END, f"[{idx}] {item['timestamp']}\n")
            text.insert(tk.END, f"源进制：{item['from_base_name']} | 输入值：{item['input_value']}\n")
            text.insert(tk.END, "转换结果：\n")
            for base, val in item['results'].items():
                base_name = self.converter.BASE_NAMES[base]
                prefix = self.converter.BASE_PREFIXES[base]
                text.insert(tk.END, f"  - {base_name}：{prefix}{val}\n")
            text.insert(tk.END, "-" * 40 + "\n\n")

        # 设置为只读
        text.config(state="readonly")

    def _clear_history(self):
        """清除历史记录按钮点击事件"""
        if messagebox.askyesno("确认", "是否确定清空所有转换历史记录？"):
            self.converter.clear_history()
            messagebox.showinfo("提示", "历史记录已清空！")

def main():
    """程序入口"""
    try:
        root = tk.Tk()
        app = BaseConverterGUI(root)
        root.mainloop()
    except Exception as e:
        messagebox.showerror("错误", f"程序运行出错：{str(e)}")

if __name__ == "__main__":
    main()


功能特点
  • 支持二进制(2)、八进制(8)、十进制(10)和十六进制(16)之间的相互转换
  • 自动验证输入格式，确保输入符合指定进制
  • 支持带前缀输入（如0b、0o、0x）
  • 历史记录功能，自动保存最近10次转换记录
  • 从历史记录加载转换结果
  • 清除历史记录功能
  • 详细的帮助信息

def to_base(dst_base: int, src_base: int, src_data: str) -> str:

    """

    dst_base: 目标进制(2-62)

    src_base: 来源进制(2-62)

    src_data: 来源字符串

    Returns:

        目标进制的字符串(如果明确指明了正负号, 就保留正负号, 例如'-0')

    Raises:

        ValueError: 如果进制超出范围或数据无效

    """

    digits62: str = "0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz"

    sign: str = ''

    if len(src_data) and (src_data[0] in '-+'):

        sign, src_data = src_data[0], src_data[1:]

    cache: list[str] = []

    num: int = 0

    src_data: str = src_data.upper() if src_base <= 36 else src_data

    for char in src_data:

        index: int = digits62.find(char)

        if index == -1 or src_base <= index:

            raise ValueError(f"无效字符:'{char}'")

        num = num * src_base + index

    while 0 < num:

        num, remainder = divmod(num, dst_base)

        cache.append(digits62[remainder])

    if not cache and len(src_data):

        cache.append(str(0))

    if sign:  # if sign == '-':  # 要显示正负值,

        cache.append(sign)

    return ''.join(reversed(cache))
